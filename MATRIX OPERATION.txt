def add_matrix(matrix1,matrix2):
    rows=len(matrix1)
    cols=len(matrix1[0])
    result=[[matrix1[i][j]+matrix2[i][j] for j in range(cols)] for i in range(rows)]
    print("the result of 2 matrix")
    return result
matrix=add_matrix(matrix1,matrix2)
for rows in matrix:
    print(rows)
